The SMIL declaration

n  <button class="time" end="c.click" id="c">
n    Click to launch!
n  </button>
n  <SMIL:par end="clip.click" id="example">
n    <span class="time">Launching!</span>
n    <SMIL:video src="spaceshuttle.avi" id="clip">

n                    dur="5" to="true"
n                    attributeName="mute"/>
n    </SMIL:video>
n    <SMIL:seq begin="rewind.click" dur="0"
n              onend="rewind(example);"/>
n  </SMIL:par>
n</SMIL:seq>

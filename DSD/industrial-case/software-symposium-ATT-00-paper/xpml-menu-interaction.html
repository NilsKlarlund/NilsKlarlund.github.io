<?include "xpml-core.dsd" ?>
<!-- define some variations on how a menu is interpreted -->
StringTypeDef ID=Menu-interaction-name:
("basic" | "optional" | "confirmation" | "timeoutconfirmation")
<!-- Add an interaction attribute to menu-constraint -->
ConstraintDef RenewID=menu-constraint:
Constraint menu-constraint[Current];
interaction="Menu-interaction-name" [Optional];
<!-- Some values of the interaction attribute entail that more message elements are possible -->
If {Attribute interaction="basic" Or Attribute interaction="optional"}
Then {
<counttimeout>
Constraint message-attributes;
Content menu-message-content
</counttimeout>[Defaultable];
<pause>
Constraint message-attributes;
Content menu-message-content
</pause>[Defaultable]}
<!-- Here are the defaults for the new messages -->
Default
select,
menu {
<help>No help is available.</help>;
<initial>
<enumerate>
<option/>
Press
<emph><dtmf/></emph>
.
</enumerate>
</initial>;
<toomanyerrors>Sorry, too many errors.</toomanyerrors>;
<timeout>You have exceeded the time limit.</timeout>;
<toomanyerrors>Sorry, too many errors.</toomanyerrors>;
<counttimeout>Sorry, too many timeouts.</counttimeout>;
<pause> Pausing. Press pound sign to continue. </pause>}
<!-- Also, there's a default for the interaction attribute itself -->
Default
select,
menu {
interaction:"basic"}
